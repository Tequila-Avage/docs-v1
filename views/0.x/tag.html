    # 在顶部导航中添加一个按钮

    // change /docs/.vuepress/config.js

module.exports = {
  theme: 'reco',
  themeConfig: {
    nav: [
      { text: 'Tags', link: '/tags/' }
    ]
  }    
}

    # 添加所需的文件

    /docs/tags/README.md

    ---
isTags: true
sidebar: false
isComment: false
---
